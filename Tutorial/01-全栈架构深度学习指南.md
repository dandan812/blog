### 2.1 Composable 模式（核心）

**阅读文件：** `apps/web/composables/usePosts.ts`

这个文件是**前端架构的核心**，展示了如何封装 API 调用：

```typescript
// 关键设计模式：
export const usePosts = () => {
  // 1. 响应式状态管理
  const posts = ref<Post[]>([])
  const pending = ref(false)
  const error = ref<string | null>(null)

  // 2. 异步数据获取
  const fetchPosts = async (params) => {
    pending.value = true
    try {
      const res = await fetchApi<PostsResponse>(`/posts?${query}`)
      posts.value = res.data
      return res
    } catch (e) {
      error.value = e.message
    } finally {
      pending.value = false
    }
  }

  // 3. 返回响应式数据和方法
  return { posts, pending, error, fetchPosts }
}
```

**学习重点：**
- ✅ 为什么要封装成 composable？（复用、测试、维护）
- ✅ 如何处理加载状态？（pending）
- ✅ 如何处理错误？（try-catch + error state）
- ✅ 如何设计 API 类型？（TypeScript 泛型）

**渲染模式对比：**

| 模式 | 说明 | 适用场景 | 配置 |
|------|------|----------|------|
| SSR | 服务端渲染，每次请求都生成 HTML | 动态内容、需要实时数据 | 默认 |
| SSG | 静态生成，构建时生成 HTML | 内容不常变化 | `prerender: true` |
| SWR | 缓存优先，定期重新验证 | 内容偶尔更新 | `swr: 3600` |
| CSR | 客户端渲染 | 需要客户端状态 | `ssr: false` |

### 3.3 控制器层设计

**阅读文件：** `apps/server/src/controllers/postController.ts`

```typescript
export const getPosts = async (req, res) => {
  // 1. 参数验证
  const { page = 1, pageSize = 10, published } = req.query

  // 2. 数据库查询
  const posts = await prisma.post.findMany({
    where: { published: published === 'true' },
    include: { tags: true, author: true },
    orderBy: { createdAt: 'desc' },
    skip: (page - 1) * pageSize,
    take: pageSize,
  })

  // 3. 响应格式化
  res.json({
    data: posts,
    total,
    page,
    pageSize,
    totalPages: Math.ceil(total / pageSize),
  })
}
```

**学习重点：**
- ✅ 控制器职责（参数验证、业务逻辑、响应格式化）

- ✅ 响应数据结构设计

```typescript
import { Request, Response } from 'express'
import prisma from '../utils/db'

export const getCategories = async (req: Request, res: Response) => {
  try {
    const categories = await prisma.category.findMany({
      include: {
        _count: {
          select: { posts: true }
        }
      }
    })
    res.json(categories)
  } catch (error) {
    res.status(500).json({ error: '获取分类失败' })
  }
}

export const getCategoryPosts = async (req: Request, res: Response) => {
  try {
    const { slug } = req.params
    const { page = 1, pageSize = 10 } = req.query

    const category = await prisma.category.findUnique({
      where: { slug },
      include: {
        posts: {
          where: { published: true },
          include: { tags: true, author: true },
          orderBy: { createdAt: 'desc' },
          skip: (Number(page) - 1) * Number(pageSize),
          take: Number(pageSize)
        }
      }
    })

    if (!category) {
      return res.status(404).json({ error: '分类不存在' })
    }

    res.json(category)
  } catch (error) {
    res.status(500).json({ error: '获取分类文章失败' })
  }
}
```

#### **步骤 3：前端 Composable**

创建 `apps/web/composables/useCategories.ts`：

```typescript
import type { Ref } from 'vue'

export interface Category {
  id: string
  name: string
  slug: string
  _count?: {
    posts: number
  }
}

export const useCategories = () => {
  const categories: Ref<Category[]> = ref([])
  const pending = ref(false)
  const error = ref<string | null>(null)

  const fetchCategories = async () => {
    pending.value = true
    error.value = null

    try {
      const apiBase = useRuntimeConfig().public.apiBaseUrl
      const response = await fetch(`${apiBase}/api/categories`)
      
      if (!response.ok) {
        throw new Error('获取分类失败')
      }

      categories.value = await response.json()
    } catch (e) {
      error.value = e instanceof Error ? e.message : '加载失败'
    } finally {
      pending.value = false
    }
  }

  return { categories, pending, error, fetchCategories }
}
```
