// usePosts.ts 核心模式
export const usePosts = () => {
  const posts = ref<Post[]>([])
  const pending = ref(false)
  const error = ref<string | null>(null)

  const fetchPosts = async (params) => {
    pending.value = true
    try {
      const res = await fetchApi<PostsResponse>(`/posts?${query}`)
      posts.value = res.data
      return res
    } catch (e) {
      error.value = e.message
    } finally {
      pending.value = false
    }
  }

  return { posts, pending, error, fetchPosts }
}
```

// postController.ts 核心模式
export const getPosts = async (req: Request, res: Response) => {
  try {
    // 1. 参数验证
    const { page = 1, pageSize = 10, published } = req.query

    // 2. 数据库查询
    const posts = await prisma.post.findMany({
      where: { published: published === 'true' },
      include: { tags: true, author: true },
      orderBy: { createdAt: 'desc' },
      skip: (Number(page) - 1) * Number(pageSize),
      take: Number(pageSize),
    })

    const total = await prisma.post.count({
      where: { published: published === 'true' }
    })

    // 3. 响应格式化
    res.json({
      data: posts,
      total,
      page: Number(page),
      pageSize: Number(pageSize),
      totalPages: Math.ceil(total / Number(pageSize)),
    })
  } catch (error) {
    res.status(500).json({ error: '获取文章失败' })
  }
}
```

```
1. 用户访问 /blog
   ↓
2. pages/blog/index.vue 渲染
   ↓
3. onMounted 调用 fetchPosts()
   ↓
4. composables/usePosts.ts 的 fetchPosts()
   ↓
5. fetchApi() 发起 HTTP 请求
   ↓
6. 后端 routes/posts.ts 接收请求
   ↓
7. controllers/postController.ts 的 getPosts()
   ↓
8. Prisma 查询数据库
   ↓
9. 返回 JSON 响应
   ↓
10. 前端更新 posts.value
   ↓
11. Vue 响应式更新 DOM
```
